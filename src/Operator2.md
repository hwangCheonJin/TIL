Operator(비교 연산자)
=============
```java
public class Operator2 {
    public static void main(String[] args) {
        int i = 10;
        int j = 10;

        //true 혹은 flase값을 반환
        System.out.println(i == j); //같다
        System.out.println(i != j); //다르다
        System.out.println(i < j); // j가 i보다 크다
        System.out.println(i <= j); // j가 i보다 크거나 같다
        System.out.println(i > j); // i가 j보다 크다
        System.out.println(i >= j); // i가 j보다 크거나 같다

        i += 10;	//i = i + 10;

        System.out.println(i);
    }
}
```
