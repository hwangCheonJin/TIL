Variable(변수)
=============
```java
    public static void main(String[] args) {
        int count;

        count = 10;

        count = 20;

//		정수는 마지막 값이 담긴다.

        double avg = 11.1;

        String name = "Jami";


        System.out.println(count);
        System.out.println(avg);
        System.out.println(name);
    }
```
